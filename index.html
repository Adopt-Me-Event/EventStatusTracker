async function updateDashboard() { const res = await fetch('/data'); const allPlayers = await res.json(); const container = document.getElementById('stats-container'); container.innerHTML = ''; // Clear old data for (let name in allPlayers) { const p = allPlayers[name]; container.innerHTML += `

${name}

Bucks: ${p.bucks.toLocaleString()}

Gingerbread: ${p.gingerbread.toLocaleString()}

Sleighball: ${p.sleighball}

Last Sync: ${p.lastUpdate}

`; } } setInterval(updateDashboard, 1000);
